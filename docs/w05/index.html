invert_rescale_gini <- function(scaled_vals, old_min, old_max, new_min, new_max) {
  old_min <- 0.348
  old_max <- 0.462
  new_min <- 0.5
  new_max <- 0.9
  inv_factor <- (scaled_vals - new_min) / (new_max - new_min)
  return(
    inv_factor * (old_max - old_min) + old_min
  )
}
